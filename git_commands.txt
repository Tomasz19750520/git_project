git log  - listuje zmiany
git status - pokazuje status
git add .  - dodaje wszystkie nowe i zmienione  pliki do stage'a
git commit -m "opis zmiany"
git restore --staged todo.txt  - anuluje ostatnie zmiany
git checkout .  - przywroc stan ze stage'a 
git reset --hard - sync folder z repozytorium

git branch nowy_branch  - tworzy nowy branch
git checkout nowy_branch  - przenosi nas do nowego brancha
git switch nowy_branch - przenosi nas do nowego brancha
git checkout -b nowy_branch  - tworzy i przenosi nas do utworzonego brancha 

git log --oneline  - pokazuje wszystkie commity
git reset --soft HEAD~1  - uncommituje ostatnia zmiane

git merge create-navigation ( scala branch z glownym main )

git branch -d create-navigation	

#GITHUB

create .ssh folder
create ssh keys
ssh-keygen -t ed25519 -C "twisniewski2@gmail.com"

touch config
paste below to config file:

HOST github.com
  AddKeysToAgent yes
  UseKeychain yes
  IdentityFile ~/.ssh/id_ed25519

Create github account and login there

go to settings and add .ssh "new SSH key"

copy first .pub key to clipboard

pbcopy < ~/.ssh/id_ed25519.pub

paste to key
save
